98 static natsStatus
99 uvScheduleToEventLoop(natsLibuvEvents *nle, int eventType, bool add)
100 {
101  natsLibuvEvent *newEvent = NULL;
102  int res;
103 
104  newEvent = (natsLibuvEvent*) malloc(sizeof(natsLibuvEvent));
105  if (newEvent == NULL)
106  return NATS_NO_MEMORY;
107 
108  newEvent->type = eventType;
109  newEvent->add = add;
110  newEvent->next = NULL;
111 
112  uv_mutex_lock(nle->lock);
113 
114  if (nle->head == NULL)
115  nle->head = newEvent;
116 
117  if (nle->tail != NULL)
118  nle->tail->next = newEvent;
119 
120  nle->tail = newEvent;
121 
122  uv_mutex_unlock(nle->lock);
123 
124  res = uv_async_send(nle->scheduler);
125 
126  return (res == 0 ? NATS_OK : NATS_ERR);
127 }

146 {

160  else

162  }
163 
164  if (nle->events)
165  res = uv_poll_start(nle->handle, nle->events, natsLibuvPoll);
166  else
167  res = uv_poll_stop(nle->handle);

169  if (res != 0)
170  return NATS_ERR;
171 
172  return NATS_OK;

181 static natsStatus
182 uvAsyncAttach(natsLibuvEvents *nle)
183 {
184  natsStatus s = NATS_OK;
185 
186  // We are reconnecting, destroy the old handle, create a new one
187  if (nle->handle != NULL)
188  {
189  uv_close((uv_handle_t*) nle->handle, uvHandleClosedCb);
190  nle->handle = NULL;
191  }
192 
193  nle->handle = (uv_poll_t*) malloc(sizeof(uv_poll_t));
194  if (nle->handle == NULL)
195  s = NATS_NO_MEMORY;
196 
197  if ((s == NATS_OK)
198  && (uv_poll_init(nle->loop, nle->handle, nle->socket) != 0))
199  {
200  s = NATS_ERR;
201  }
202 
203  if ((s == NATS_OK)
204  && (nle->handle->data = (void*) nle)
205  && (uv_poll_start(nle->handle, UV_READABLE, natsLibuvPoll) != 0))
206  {
207  s = NATS_ERR;
208  }
209 
210  return s;
211 }

245 static void
246 uvAsyncCb(uv_async_t *handle)
247 {
248  natsLibuvEvents *nle = (natsLibuvEvents*) handle->data;
249  natsStatus s = NATS_OK;
250  natsLibuvEvent *event = NULL;
251  bool remove = false;
252  bool more = false;
253 
254  while (1)
255  {
256  uv_mutex_lock(nle->lock);
257 
258  event = nle->head;
259  if (event == NULL)
260  {
261  // This is possible, even on entry of this function because
262  // the callback is called when the handle is initialized.
263  uv_mutex_unlock(nle->lock);
264  return;
265  }
266 
267  nle->head = event->next;
268  if (event == nle->tail)
269  nle->tail = NULL;
270 
271  more = (nle->head != NULL ? true : false);
272 
273  uv_mutex_unlock(nle->lock);
274 
275  switch (event->type)
276  {
277  case NATS_LIBUV_ATTACH:
278  {
279  s = uvAsyncAttach(nle);
280  break;
281  }
282  case NATS_LIBUV_READ:
283  case NATS_LIBUV_WRITE:
284  {
285  s = uvPollUpdate(nle, event->type, event->add);
286  break;
287  }
288  case NATS_LIBUV_DETACH:
289  {
290  uvAsyncDetach(nle);
291  break;
292  }
293  default:
294  {
295  s = NATS_ERR;
296  break;
297  }
298  }
299 
300  free(event);
301 
302  if ((s != NATS_OK) || !more)
303  break;
304  }
305 
306  if (s != NATS_OK)
307  natsConnection_Close(nle->nc);
308 }

322 natsStatus
323 natsLibuv_Attach(void **userData, void *loop, natsConnection *nc, natsSock socket)
324 {
325  uv_loop_t *uvLoop = (uv_loop_t*) loop;
326  bool sched = false;
327  natsLibuvEvents *nle = (natsLibuvEvents*) (*userData);
328  natsStatus s = NATS_OK;
329 
330  sched = ((uv_key_get(&uvLoopThreadKey) != loop) ? true : false);
331 
332  // This is the first attach (when reconnecting, nle will be non-NULL).
333  if (nle == NULL)
334  {
335  // This has to run from the event loop!
336  if (sched)
337  return NATS_ILLEGAL_STATE;
338 
339  nle = (natsLibuvEvents*) calloc(1, sizeof(natsLibuvEvents));
340  if (nle == NULL)
341  return NATS_NO_MEMORY;
342 
343  nle->lock = (uv_mutex_t*) malloc(sizeof(uv_mutex_t));
344  if (nle->lock == NULL)
345  s = NATS_NO_MEMORY;
346 
347  if ((s == NATS_OK) && (uv_mutex_init(nle->lock) != 0))
348  s = NATS_ERR;
349 
350  if ((s == NATS_OK)
351  && ((nle->scheduler = (uv_async_t*) malloc(sizeof(uv_async_t))) == NULL))
352  {
353  s = NATS_NO_MEMORY;
354  }
355 
356  if ((s == NATS_OK)
357  && (uv_async_init(uvLoop, nle->scheduler, uvAsyncCb) != 0))
358  {
359  s = NATS_ERR;
360  }
361 
362  if (s == NATS_OK)
363  {
364  nle->nc = nc;
365  nle->loop = uvLoop;
366  nle->scheduler->data = (void*) nle;
367  }
368  }
369 
370  if (s == NATS_OK)
371  {
372  nle->socket = socket;
373  nle->events = UV_READABLE;
374 
375  if (sched)
376  s = uvScheduleToEventLoop(nle, NATS_LIBUV_ATTACH, true);
377  else
378  s = uvAsyncAttach(nle);
379  }
380 
381  if (s == NATS_OK)
382  *userData = (void*) nle;
383  else
384  natsLibuv_Detach((void*) nle);
385 
386  return s;
387 }

397 natsStatus
398 natsLibuv_Read(void *userData, bool add)
399 {
400  natsLibuvEvents *nle = (natsLibuvEvents*) userData;
401  natsStatus s = NATS_OK;
402  bool sched;
403 
404  sched = ((uv_key_get(&uvLoopThreadKey) != nle->loop) ? true : false);
405 
406  // If this call is made from a different thread than the event loop's
407  // thread, or if there are already scheduled events, then schedule
408  // this new event.
409 
410  // We don't need to get the lock for nle->head because if sched is
411  // false, we are in the event loop thread, which is the thread removing
412  // events from the list. Also, all calls to the read/write/etc.. callbacks
413  // are protected by the connection's lock in the NATS library.
414  if (sched || (nle->head != NULL))
415  s = uvScheduleToEventLoop(nle, NATS_LIBUV_READ, add);
416  else
417  s = uvPollUpdate(nle, NATS_LIBUV_READ, add);
418 
419  return s;
420 }
