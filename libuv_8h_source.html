109 static natsStatus
110 uvScheduleToEventLoop(natsLibuvEvents *nle, int eventType, bool add)
111 {
112  natsLibuvEvent *newEvent = NULL;
113  int res;
114 
115  newEvent = (natsLibuvEvent*) malloc(sizeof(natsLibuvEvent));
116  if (newEvent == NULL)
117  return NATS_NO_MEMORY;
118 
119  newEvent->type = eventType;
120  newEvent->add = add;
121  newEvent->next = NULL;
122 
123  uv_mutex_lock(nle->lock);
124 
125  if (nle->head == NULL)
126  nle->head = newEvent;
127 
128  if (nle->tail != NULL)
129  nle->tail->next = newEvent;
130 
131  nle->tail = newEvent;
132 
133  uv_mutex_unlock(nle->lock);
134 
135  res = uv_async_send(nle->scheduler);
136 
137  return (res == 0 ? NATS_OK : NATS_ERR);
138 }

140 static void
141 natsLibuvPoll(uv_poll_t* handle, int status, int events)
142 {
143  natsLibuvEvents *nle = (natsLibuvEvents*)handle->data;
144 
145  if (status != 0)
146  return;
147 
148  if (events & UV_READABLE)
149  natsConnection_ProcessReadEvent(nle->nc);
150 
151  if (events & UV_WRITABLE)
152  natsConnection_ProcessWriteEvent(nle->nc);
153 }

155 static natsStatus
156 uvPollUpdate(natsLibuvEvents *nle, int eventType, bool add)
157 {
158  int res;
159 
160  if (eventType == NATS_LIBUV_READ)
161  {
162  if (add)
163  nle->events |= UV_READABLE;
164  else
165  nle->events &= ~UV_READABLE;
166  }
167  else
168  {
169  if (add)
170  nle->events |= UV_WRITABLE;
171  else
172  nle->events &= ~UV_WRITABLE;
173  }
174 
175  if (nle->events)
176  res = uv_poll_start(nle->handle, nle->events, natsLibuvPoll);
177  else
178  res = uv_poll_stop(nle->handle);
179 
180  if (res != 0)
181  return NATS_ERR;
182 
183  return NATS_OK;
184 }

192 static natsStatus
193 uvAsyncAttach(natsLibuvEvents *nle)
194 {
195  natsStatus s = NATS_OK;
196 
197  // We are reconnecting, destroy the old handle, create a new one
198  if (nle->handle != NULL)
199  {
200  uv_close((uv_handle_t*) nle->handle, uvHandleClosedCb);
201  nle->handle = NULL;
202  }
203 
204  nle->handle = (uv_poll_t*) malloc(sizeof(uv_poll_t));
205  if (nle->handle == NULL)
206  s = NATS_NO_MEMORY;
207 
208  if (s == NATS_OK)
209  {
210 #if UV_VERSION_MAJOR <= 1
211  if (uv_poll_init_socket(nle->loop, nle->handle, nle->socket) != 0)
212 #else
213  if (uv_poll_init(nle->loop, nle->handle, nle->socket) != 0)
214 #endif
215  s = NATS_ERR;
216  }
217 
218  if ((s == NATS_OK)
219  && (nle->handle->data = (void*) nle)
220  && (uv_poll_start(nle->handle, UV_READABLE, natsLibuvPoll) != 0))
221  {
222  s = NATS_ERR;
223  }
224 
225  return s;
226 }

260 static void
261 uvAsyncCb(uv_async_t *handle)
262 {
263  natsLibuvEvents *nle = (natsLibuvEvents*) handle->data;
264  natsStatus s = NATS_OK;
265  natsLibuvEvent *event = NULL;
266  bool remove = false;
267  bool more = false;
268 
269  while (1)
270  {
271  uv_mutex_lock(nle->lock);
272 
273  event = nle->head;
274  if (event == NULL)
275  {
276  // This is possible, even on entry of this function because
277  // the callback is called when the handle is initialized.
278  uv_mutex_unlock(nle->lock);
279  return;
280  }
281 
282  nle->head = event->next;
283  if (event == nle->tail)
284  nle->tail = NULL;
285 
286  more = (nle->head != NULL ? true : false);
287 
288  uv_mutex_unlock(nle->lock);
289 
290  switch (event->type)
291  {
292  case NATS_LIBUV_ATTACH:
293  {
294  s = uvAsyncAttach(nle);
295  break;
296  }
297  case NATS_LIBUV_READ:
298  case NATS_LIBUV_WRITE:
299  {
300  s = uvPollUpdate(nle, event->type, event->add);
301  break;
302  }
303  case NATS_LIBUV_DETACH:
304  {
305  uvAsyncDetach(nle);
306  break;
307  }
308  default:
309  {
310  s = NATS_ERR;
311  break;
312  }
313  }
314 
315  free(event);
316 
317  if ((s != NATS_OK) || !more)
318  break;
319  }
320 
321  if (s != NATS_OK)
322  natsConnection_Close(nle->nc);
323 }

337 natsStatus
338 natsLibuv_Attach(void **userData, void *loop, natsConnection *nc, natsSock socket)
339 {
340  uv_loop_t *uvLoop = (uv_loop_t*) loop;
341  bool sched = false;
342  natsLibuvEvents *nle = (natsLibuvEvents*) (*userData);
343  natsStatus s = NATS_OK;
344 
345  sched = ((uv_key_get(&uvLoopThreadKey) != loop) ? true : false);
346 
347  // This is the first attach (when reconnecting, nle will be non-NULL).
348  if (nle == NULL)
349  {
350  // This has to run from the event loop!
351  if (sched)
352  return NATS_ILLEGAL_STATE;
353 
354  nle = (natsLibuvEvents*) calloc(1, sizeof(natsLibuvEvents));
355  if (nle == NULL)
356  return NATS_NO_MEMORY;
357 
358  nle->lock = (uv_mutex_t*) malloc(sizeof(uv_mutex_t));
359  if (nle->lock == NULL)
360  s = NATS_NO_MEMORY;
361 
362  if ((s == NATS_OK) && (uv_mutex_init(nle->lock) != 0))
363  s = NATS_ERR;
364 
365  if ((s == NATS_OK)
366  && ((nle->scheduler = (uv_async_t*) malloc(sizeof(uv_async_t))) == NULL))
367  {
368  s = NATS_NO_MEMORY;
369  }
370 
371  if ((s == NATS_OK)
372  && (uv_async_init(uvLoop, nle->scheduler, uvAsyncCb) != 0))
373  {
374  s = NATS_ERR;
375  }
376 
377  if (s == NATS_OK)
378  {
379  nle->nc = nc;
380  nle->loop = uvLoop;
381  nle->scheduler->data = (void*) nle;
382  }
383  }
384 
385  if (s == NATS_OK)
386  {
387  nle->socket = socket;
388  nle->events = UV_READABLE;
389 
390  if (sched)
391  s = uvScheduleToEventLoop(nle, NATS_LIBUV_ATTACH, true);
392  else
393  s = uvAsyncAttach(nle);
394  }
395 
396  if (s == NATS_OK)
397  *userData = (void*) nle;
398  else
399  natsLibuv_Detach((void*) nle);
400 
401  return s;
402 }

412 natsStatus
413 natsLibuv_Read(void *userData, bool add)
414 {
415  natsLibuvEvents *nle = (natsLibuvEvents*) userData;
416  natsStatus s = NATS_OK;
417  bool sched;
418 
419  sched = ((uv_key_get(&uvLoopThreadKey) != nle->loop) ? true : false);
420 
421  // If this call is made from a different thread than the event loop's
422  // thread, or if there are already scheduled events, then schedule
423  // this new event.
424 
425  // We don't need to get the lock for nle->head because if sched is
426  // false, we are in the event loop thread, which is the thread removing
427  // events from the list. Also, all calls to the read/write/etc.. callbacks
428  // are protected by the connection's lock in the NATS library.
429  if (sched || (nle->head != NULL))
430  s = uvScheduleToEventLoop(nle, NATS_LIBUV_READ, add);
431  else
432  s = uvPollUpdate(nle, NATS_LIBUV_READ, add);
433 
434  return s;
435 }
